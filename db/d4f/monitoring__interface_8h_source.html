32enum class metric_type {

42inline std::string to_string(metric_type type) {
43 switch(type) {
44 case metric_type::gauge: return "GAUGE";
45 case metric_type::counter: return "COUNTER";
46 case metric_type::histogram: return "HISTOGRAM";
47 case metric_type::summary: return "SUMMARY";
48 default: return "UNKNOWN";
49 }
50}

55inline Result<metric_type> metric_type_from_string(const std::string& str) {
56 std::string upper = str;
57 std::transform(upper.begin(), upper.end(), upper.begin(),
58 [](unsigned char c) { return static_cast<char>(std::toupper(c)); });
59
60 if (upper == "GAUGE") return ok(metric_type::gauge);
61 if (upper == "COUNTER") return ok(metric_type::counter);
62 if (upper == "HISTOGRAM") return ok(metric_type::histogram);
63 if (upper == "SUMMARY") return ok(metric_type::summary);
64
65 return error<metric_type>(1, "Invalid metric type: " + str, "monitoring_interface");
66}
