196template<typename T, typename F>
197auto map(const Result<T>& result, F&& func)
198 -> Result<decltype(func(std::declval<T>()))> {
199 using ReturnType = decltype(func(std::declval<T>()));

201 if (is_ok(result)) {

264template<typename T, typename F>
265Result<T> try_catch(F&& func, const std::string& module = "") {
266 try {
267 if constexpr (std::is_void_v<T>) {
268 func();
269 return ok();
270 } else {
271 return ok<T>(func());
272 }
273 } catch (const std::exception& e) {
274 return error<T>(error_codes::INTERNAL_ERROR, e.what(), module);
275 } catch (...) {
276 return error<T>(error_codes::INTERNAL_ERROR, "Unknown error", module);
277 }
278}
