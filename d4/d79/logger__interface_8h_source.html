224inline std::string to_string(log_level level) {
225 switch(level) {
226 case log_level::trace: return "TRACE";
227 case log_level::debug: return "DEBUG";
228 case log_level::info: return "INFO";
229 case log_level::warning: return "WARNING";
230 case log_level::error: return "ERROR";
231 case log_level::critical: return "CRITICAL";
232 case log_level::off: return "OFF";
233 default: return "UNKNOWN";
234 }
235}

240inline log_level from_string(const std::string& str) {
241 std::string upper = str;
242 std::transform(upper.begin(), upper.end(), upper.begin(), ::toupper);
243
244 if (upper == "TRACE") return log_level::trace;
245 if (upper == "DEBUG") return log_level::debug;
246 if (upper == "INFO") return log_level::info;
247 if (upper == "WARNING" || upper == "WARN") return log_level::warning;
248 if (upper == "ERROR") return log_level::error;
249 if (upper == "CRITICAL" || upper == "FATAL") return log_level::critical;
250 if (upper == "OFF") return log_level::off;
251 return log_level::info; // default
252}
