157inline std::string to_string(log_level level) {
158 switch(level) {
159 case log_level::trace: return "TRACE";
160 case log_level::debug: return "DEBUG";
161 case log_level::info: return "INFO";
162 case log_level::warning: return "WARNING";
163 case log_level::error: return "ERROR";
164 case log_level::critical: return "CRITICAL";
165 case log_level::off: return "OFF";
166 default: return "UNKNOWN";
167 }

173inline log_level from_string(const std::string& str) {
174 if (str == "TRACE") return log_level::trace;
175 if (str == "DEBUG") return log_level::debug;
176 if (str == "INFO") return log_level::info;
177 if (str == "WARNING") return log_level::warning;
178 if (str == "ERROR") return log_level::error;
179 if (str == "CRITICAL") return log_level::critical;
180 if (str == "OFF") return log_level::off;
181 return log_level::info; // default
